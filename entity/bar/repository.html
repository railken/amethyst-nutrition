Railken\Amethyst\Repositories\BarRepository

The repository is the class to perform queries.

use Railken\Amethyst\Managers\BarManager;

$manager = new BarManager();

$repository = $manager->getRepository();

Retrieving an entity

$repository->findOneBy(['id' => 1]);
$repository->findOneById(1);

Retrieving all entities

$repository->findAll();

Performing a query using \Illuminate\DataBase\Eloquent\Builder

$repository->newQuery()->where('id', 1)->first();

Extend the class

Create the new repository in app/Repositories/BarRepository

namespace App\Repositories;

use Railken\Amethyst\Repositories\BarRepository as Repository;

class BarRepository extends Repository {
    // ...
}

Update the file configs/amethyst.foo with the new class

return [
    'data' => [
        'bar' => [
            'repository' => App\Repositories\BarRepository::class,
        ],
    ]
];
